		1.3.25
		1.3.26		PROCEDURE isHovering(THIS, PROCEDURE draw, PROCEDURE click, BOOLEAN effectState) RETURNS NONE # If mouse is hovering
		1.3.27			# Checks to see if the mouse is hovering over a button
		1.3.28			IF THIS.face.collidepoint(pygame.mouse.get_pos()) THEN # If the button is not already being hovered
		1.3.29				IF NOT THIS.hovering # Sets button being hovered to true
		1.3.30					SET THIS.hovering AS TRUE
		1.3.31					THIS.text.ChangeColor((255, 0, 0))
		1.3.32					pygame.mixer.Channel(0).set_volume(.3) IF effectState ELSE pygame.mixer.Channel(0).set_volume(0) END IF
		1.3.33					draw()
		1.3.34				
		1.3.35				ELSE # If the button is already being hovered
		1.3.36					THIS.HasClicked(click) # Check if the button has been clicked
		1.3.37				END IF
		1.3.38
		1.3.39			# If mouse is not over the button but the button is still being hovered    
		1.3.40			ELSE IF NOT THIS.face.collidepoint(pygame.mouse.get_pos()) AND THIS.hovering THEN
		1.3.41				SET THIS.hovering AS FALSE
		1.3.42				SET THIS.text.ChangeColor((255, 255, 255))
		1.3.43				draw()
		1.3.44			END IF
		1.3.45		
		1.3.46		PROCEDURE HasClicked(THIS, PROCEDURE click) RETURNS NONE
		1.3.47			# Check if a button has been clicked
		1.3.48			IF pygame.mouse.get_pressed()[0] THEN
		1.3.49				pygame.mixer.Channel(0).play(pygame.mixer.Sound(THIS.selectSound))
		1.3.50				click()
		1.3.51				time.sleep(.1) # Stop multiple clicks being allowed
		1.3.52			END IF
		1.3.53			
		1.3.54		PROCEDURE RenderBtn(THIS) RETURNS NONE
		1.3.55			pygame.draw.rect(pygame.display.get_surface(), (0, 0, 0), THIS.face)
		1.3.56			THIS.text.RenderText()
		1.3.57		END PROCEDURE

		1.3.59		FUNCTION __str__(THIS) RETURNS STRING
		1.3.60			RETURN 'Btn' # Returns only when str() function called
